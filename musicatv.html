<script>

  // Obtén el elemento que quieres hacer fullscreen

  const elementoFullscreen = document.documentElement;


  // Solicita entrar en modo fullscreen

  if (elementoFullscreen.requestFullscreen) {

    elementoFullscreen.requestFullscreen();

  } else if (elementoFullscreen.mozRequestFullScreen) { // Firefox

    elementoFullscreen.mozRequestFullScreen();

  } else if (elementoFullscreen.webkitRequestFullscreen) { // Chrome, Safari y Opera

    elementoFullscreen.webkitRequestFullscreen();

  } else if (elementoFullscreen.msRequestFullscreen) { // IE/Edge

    elementoFullscreen.msRequestFullscreen();

  }

</script>
